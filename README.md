var day = "tuesday"
switch(day)
{
    case "monday":
     console.log("chest day")
    break
    case "tuesday":
     console.log("bi-sep & ti-sep day")
    break
    case "wednesday":
     console.log("sholder day")
    break
    case "thursday":
     console.log("back day")
    break
    case "friday":
     console.log("abs day")
    break
    case "saturday":
     console.log("leg day")
    break
    case "sunday":
     console.log("RESTING DAY")
    break
    default:console.log("day does not exist")
}
